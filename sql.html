ORACLE SQL

مرجع SQL العملي الكامل

الأوامر مرتبة من إنشاء البنية إلى تعديل البيانات والصلاحيات والكائنات والربط.

1. SQL Basics

SELECTWHEREORDER BYDISTINCTALIASESNULL
SELECT employee_id, first_name, salary
FROM employees
WHERE salary > 5000
ORDER BY salary DESC;

2. DDL

تغيّر بنية قاعدة البيانات.

CREATEALTERDROPTRUNCATERENAMECOMMENT
CREATE TABLE products (
 pid NUMBER(8) PRIMARY KEY,
 pname VARCHAR2(25) NOT NULL,
 price NUMBER(8,2) CHECK(price >= 0)
);
ALTER TABLE products ADD category VARCHAR2(30);
ALTER TABLE products MODIFY pname VARCHAR2(50);
ALTER TABLE products RENAME COLUMN category TO type;
ALTER TABLE products DROP COLUMN type;
TRUNCATE TABLE products;
RENAME products TO items;
COMMENT ON TABLE items IS 'Products list';
DROP TABLE items;

3. DML

تضيف وتعدّل وتحذف السجلات.

INSERTUPDATEDELETEMERGE
INSERT INTO products(pid,pname,price)
VALUES(1,'Laptop',3500);
UPDATE products SET price=3300 WHERE pid=1;
DELETE FROM products WHERE pid=1;

4. DCL

تتحكم في الصلاحيات.

GRANTREVOKE
GRANT CREATE SESSION, CREATE TABLE TO ali;
GRANT SELECT, INSERT ON hr.employees TO ali;
REVOKE INSERT ON hr.employees FROM ali;
REVOKE CREATE TABLE FROM ali;

5. TABLES & CONSTRAINTS

PRIMARY KEYFOREIGN KEYNOT NULLUNIQUECHECKDEFAULT
CREATE TABLE orders (
 order_id NUMBER PRIMARY KEY,
 customer_id NUMBER,
 order_date DATE DEFAULT SYSDATE,
 total NUMBER CHECK(total >= 0),
 CONSTRAINT fk_order_customer
 FOREIGN KEY(customer_id) REFERENCES customers(customer_id)
);

6. VIEW

CREATE VIEWREPLACECHECK OPTIONREAD ONLY
CREATE OR REPLACE VIEW v50 AS
SELECT employee_id, first_name, department_id
FROM employees
WHERE department_id=50
WITH CHECK OPTION;

7. SEQUENCE

NEXTVALCURRVALCACHECYCLE
CREATE SEQUENCE s1 START WITH 1 INCREMENT BY 1 NOCYCLE;
INSERT INTO products(pid,pname) VALUES(s1.NEXTVAL,'Phone');
SELECT s1.CURRVAL FROM dual;

8. SYNONYM & INDEX

PRIVATE SYNONYMPUBLIC SYNONYMINDEXUNIQUE INDEX
CREATE SYNONYM emp FOR hr.employees;
CREATE INDEX idx_emp_name ON employees(last_name);
CREATE UNIQUE INDEX idx_emp_email ON employees(email);

9. INNER JOIN

SELECT e.first_name, d.department_name
FROM employees e
INNER JOIN departments d
ON e.department_id=d.department_id;

10. CROSS JOIN

SELECT e.first_name, d.department_name
FROM employees e
CROSS JOIN departments d;

11. GROUPING

GROUP BYHAVINGCOUNTSUMAVG
SELECT department_id, COUNT(*) total_employees
FROM employees
GROUP BY department_id
HAVING COUNT(*) >= 5;
